  PARTITION BY user_id
  ORDER BY ts
  MEASURES val AS first_val
  ONE ROW PER MATCH
  AFTER MATCH SKIP PAST LAST ROW
  PATTERN (A B*)
  DEFINE
    A AS val < 50,
    B AS val >= 50
);
